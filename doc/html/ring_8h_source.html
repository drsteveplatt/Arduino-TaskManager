124 template<class T>void ring<T>::push_front(T val) {
125  _ringNode<T>* newNode;
126  newNode = new _ringNode<T>();
127  newNode->m_val = val;
128  if(m_cur==NULL) {
129  newNode->m_next = newNode;
130  newNode->m_prev = newNode;

132  newNode->m_next = m_cur;

137  m_cur = newNode;

160  _ringNode<T>* toGo; // node being deleted

162  toGo = m_cur;
163  if(m_cur=m_cur->m_next) { m_cur=NULL; }
