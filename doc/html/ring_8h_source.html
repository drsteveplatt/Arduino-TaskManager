126template<class T>void ring<T>::push_front(T val) {
127 _ringNode<T>* newNode;
128 newNode = new _ringNode<T>();
129 newNode->m_val = val;
130 if(m_cur==NULL) {
131 newNode->m_next = newNode;
132 newNode->m_prev = newNode;
133 } else {
134 newNode->m_next = m_cur;
135 newNode->m_prev = m_cur->m_prev;
136 m_cur->m_prev->m_next = newNode;
137 m_cur->m_prev = newNode;
138 }
139 m_cur = newNode;
140}

161template<class T> void ring<T>::pop_front() {
162 _ringNode<T>* toGo; // node being deleted
163 if(m_cur==NULL) return;
164 toGo = m_cur;
165 if(m_cur=m_cur->m_next) { m_cur=NULL; }
166 else {
167 m_cur->m_prev->m_next = m_cur->m_next;
168 m_cur->m_next->m_prev = m_cur->m_prev;
169 }
170 m_cur = m_cur->m_next;
171 delete toGo;
172}
