186template<class T> void ring<T>::pop_front() {
187 _ringNode<T>* toGo; // node being deleted
188 if(m_cur==NULL) return;
189 toGo = m_cur;
190 if(m_cur=m_cur->m_next) { m_cur=NULL; }
191 else {
192 m_cur->m_prev->m_next = m_cur->m_next;
193 m_cur->m_next->m_prev = m_cur->m_prev;
194 }
195 m_cur = m_cur->m_next;
196 delete toGo;
197}
