113 template<class T>void ring<T>::push_front(T val) {
114  _ringNode<T>* newNode;
115  newNode = new _ringNode<T>();
116  newNode->m_val = val;

148 template<class T> void ring<T>::pop_front() {
149  _ringNode<T>* toGo; // node being deleted
150  if(m_cur==NULL) return;
151  toGo = m_cur;
152  if(m_cur=m_cur->m_next) { m_cur=NULL; }

155  m_cur->m_next->m_prev = m_cur->m_prev;
156  }
157  m_cur = m_cur->m_next;
158  delete toGo;
