10// Read from a stream into a String
11// This is useful in TM_Macro code, so it will need TM_BEGIN() and TM_END().
12// stream_in is the stream to use (normally Serial)
13// string_out is the string variable to place the results in
14// yield_id is the TM_MACRO ID for TM_YIELD
15// echo is a boolean -- if true, echo the characters back to stream_in as read in
16// TM_READLINE(stream_in, string_out, yield_id, echo)
17#define TM_READLINE(stream_in, string_out, yield_id, echo) \
18 { /* read in a whole line of chars into cmd */ \
19 char ch; \
20 bool done; \
21 bool lastWasCr; \
22 cmd = ""; \
23 done = false; \
24 lastWasCr = false; \
25 while(!done) { \
26 while(!stream_in.available()) { TM_YIELD(yield_id); } \
27 ch = stream_in.read(); \
28 /* Backspace processing for PuTTY */ \
29 if(ch==0x08 && echo) { \
30 if(cmd.length()>0) { \
31 Serial << '\b' << ' ' << '\b'; \
32 cmd.remove(cmd.length()-1); \
33 } \
34 } else { \
35 /* CR processing for PuTTY */ \
36 if(ch=='\r') { ch='\n'; lastWasCr = true; } \
37 else if(ch=='\n' && lastWasCr) { lastWasCr=false; continue; } /* crlf, skip the lf */ \
38 if(ch=='\n') { Serial << ch; lastWasCr = false; done = true; } \
39 else { \
40 if(echo)Serial << ch; \
41 string_out += ch; \
42 lastWasCr = false; \
43 } /* end if(ch=='\n') else */ \
44 } /* end if(ch==0x08) else */ \
45 } /* end while */ \
46 } /* end read in a whole line of chars */
